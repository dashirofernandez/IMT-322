import RPi.GPIO as GPIO
import time

# Configurar el modo de los pines GPIO
GPIO.setmode(GPIO.BCM)  # Usa la numeraci칩n BCM
GPIO.setwarnings(False)

# Configuraci칩n del pin
led_pin = 17  # Pin GPIO que est치s usando
GPIO.setup(led_pin, GPIO.OUT)

try:
    while True:
        GPIO.output(led_pin, GPIO.HIGH)  # Enciende el LED
        time.sleep(1)  # Espera 1 segundo
        GPIO.output(led_pin, GPIO.LOW)   # Apaga el LED
        time.sleep(1)  # Espera 1 segundo
except KeyboardInterrupt:
    # Limpia la configuraci칩n GPIO cuando se interrumpe el programa
    GPIO.cleanup()
